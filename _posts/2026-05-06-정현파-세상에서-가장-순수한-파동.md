소리가 공기를 타고 흐를때, 전류가 전선 위를 달릴 때, 빛이 우주를 가로질러 도달할때 그 모두의  밑바닥에는 하나의 단순한 곡선이 있다. 바로 정현파 (sinusoidal wave)이다. 정현파는 사인파 (sin wave), 코사인파(cosine wave)를 총칭하는 말이다. 푸리에 분석에 의하면 세상의 모든 복잡한 파형은 결국 다양한 정현파들의 합으로 표현된다. 정현파의 본질은 원 위의 회전에 관한것이다. 그리고, 시간에 따른 회전을 기술한것이다.

기본 수식

정현파는 네가지 파라미터로 기술된다.

A: 진폭 (파형의 최대 높이. 에너지의 세기와 직결된다.)

f: 주파수(1초에 몇 번 반복되는가. Hz)

T = 1/f : 주기(한 번 완전히 반복되는데 걸리는 시간)

φ: 위상 (파형 시작 위치. 라디안 단위)

const DPR = window.devicePixelRatio || 1;

function setupCanvas(id, w, h) {
const c = document.getElementById(id);
c.width = w * DPR; c.height = h * DPR;
c.style.width = w + 'px'; c.style.height = h + 'px';
const ctx = c.getContext('2d');
ctx.scale(DPR, DPR);
return ctx;
}

function getColor(v) {
const s = getComputedStyle(document.documentElement);
return s.getPropertyValue(v).trim();
}

function drawCircle(ctx, W, H, r, angleRad, color, showRef) {
const cx = W/2, cy = H/2;
ctx.clearRect(0, 0, W, H);
ctx.strokeStyle = getColor('--color-border-tertiary');
ctx.lineWidth = 0.5;
ctx.beginPath(); ctx.moveTo(0, cy); ctx.lineTo(W, cy); ctx.stroke();
ctx.beginPath(); ctx.moveTo(cx, 0); ctx.lineTo(cx, H); ctx.stroke();

if (showRef) {
ctx.strokeStyle = '#AFA9EC';
ctx.globalAlpha = 0.4; ctx.lineWidth = 0.5;
ctx.setLineDash([3,3]);
ctx.beginPath(); ctx.arc(cx, cy, Math.min(cx,cy)0.85, 0, 2Math.PI); ctx.stroke();
ctx.setLineDash([]); ctx.globalAlpha = 1;
}

ctx.strokeStyle = color;
ctx.lineWidth = 1.5;
ctx.beginPath(); ctx.arc(cx, cy, r, 0, 2*Math.PI); ctx.stroke();

const px = cx + r * Math.cos(-angleRad);
const py = cy + r * Math.sin(-angleRad);

ctx.strokeStyle = color; ctx.globalAlpha = 0.35; ctx.lineWidth = 1;
ctx.beginPath(); ctx.moveTo(cx, cy); ctx.lineTo(px, cy); ctx.stroke();
ctx.globalAlpha = 1;

ctx.beginPath(); ctx.moveTo(px, cy); ctx.lineTo(px, py); ctx.stroke();

}

function drawWave(ctx, W, H, f, A, refColor, mainColor) {
ctx.clearRect(0, 0, W, H);
const midY = H/2, scaleY = (H/2)*0.8;

ctx.strokeStyle = getColor('--color-border-tertiary');
ctx.lineWidth = 0.5; ctx.setLineDash([4,4]);
ctx.beginPath(); ctx.moveTo(0, midY); ctx.lineTo(W, midY); ctx.stroke();
ctx.setLineDash([]);

if (refColor) {
ctx.strokeStyle = refColor; ctx.globalAlpha = 0.35; ctx.lineWidth = 1;
ctx.beginPath();
for (let x=0; x<=W; x++) {
const t = (x/W)4;
const y = midY - scaleY * Math.sin(2Math.PI*t);
x===0 ? ctx.moveTo(x,y) : ctx.lineTo(x,y);
}
ctx.stroke(); ctx.globalAlpha = 1;
}

ctx.strokeStyle = mainColor; ctx.lineWidth = 2;
ctx.beginPath();
for (let x=0; x<=W; x++) {
const t = (x/W)4;
const y = midY - A * scaleY * Math.sin(2Math.PIft);
x===0 ? ctx.moveTo(x,y) : ctx.lineTo(x,y);
}
ctx.stroke();

ctx.fillStyle = getColor('--color-text-secondary');
ctx.font = '10px ' + getColor('--font-mono');
ctx.fillText('A='+A.toFixed(1), 6, midY - AscaleY - 5);
ctx.fillText('-A='+(-A).toFixed(1), 6, midY + AscaleY + 13);

const periodPx = W / (4*f);
if (periodPx > 30) {
ctx.strokeStyle = '#BA7517'; ctx.lineWidth = 0.5; ctx.setLineDash([3,3]);
ctx.beginPath(); ctx.moveTo(periodPx, 0); ctx.lineTo(periodPx, H); ctx.stroke();
ctx.setLineDash([]);
ctx.fillStyle = '#BA7517';
ctx.fillText('T', periodPx+3, 14);
}
}
